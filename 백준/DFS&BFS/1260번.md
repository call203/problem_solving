```
const fs = require('fs')
const filePath = process.platform === 'linux' ? '/dev/stdin' : `./input.txt`
const input = fs.readFileSync(filePath).toString().trim().split('\n')
let [N, M, V] = input[0].split(' ').map(Number)
let arr = input.slice(1, input.length).map((i) => i.split(' '))
const graph = Array.from({ length: N + 1 }, () => [])
arr.forEach(([a, b]) => {
  graph[Number(a)].push(Number(b))
  graph[Number(b)].push(Number(a))
})
/** Sorted for increasing order*/
graph.map((i) => i.sort((a, b) => a - b))

let visited = Array.from({ length: N + 1 }, () => 0)
const dfsResult = []

/** DFS */
const dfs = (node) => {
  visited[node] = 1
  dfsResult.push(node)
  graph[node].forEach((i) => {
    if (!visited[i]) {
      dfs(i)
    }
  })
}

const visited2 = Array.from({ length: N + 1 }, () => 0)

const bfsResult = []
/**BFS */
const bfs = (start) => {
  const queue = [start]

  while (queue.length) {
    let node = queue.shift()
    if (!visited2[node]) {
      visited2[node] = 1
      bfsResult.push(node)
      queue.push(...graph[node])
    }
  }
}

dfs(V)
console.log(dfsResult.join(' '))
bfs(V)
console.log(bfsResult.join(' '))
```

## BFS
### Accepted
```
const fs = require('fs')
const filePath = process.platform === 'linux' ? '/dev/stdin' : `./input.txt`
const input = fs.readFileSync(filePath).toString().trim().split('\n')
let [N] = input[0].split(' ').map(Number)
let [C1, C2] = input[1].split(' ').map(Number)
const M = Number(input[2])
const arr = input.slice(3).map((i) => i.split(' ').map(Number))

const graph = Array.from({ length: N + 1 }, () => [])
arr.forEach(([a1, a2]) => {
  graph[a1].push(a2)
  graph[a2].push(a1)
})

const visited = new Array(N + 1).fill(0)
const bfs = (one, two) => {
  let queue = [[one, 0]]

  while (queue.length) {
    let [node, count] = queue.shift()

    if (node === two) {
      return count
    }

    for (let next of graph[node]) {
      if (!visited[next]) {
        visited[next] = 1
        queue.push([next, count + 1])
      }
    }
  }
  return -1
}

console.log(bfs(C1, C2))
```
