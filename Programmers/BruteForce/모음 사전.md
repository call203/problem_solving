function solution(letter) {

    let words = ["A","E","I","O","U"]
    let answer = 0;
    let count = 0;
    const dfs = (str)=>{
        if(str ===letter){
            answer = count
            return 
        }
        if(str.length > words.length) return 
        count+=1
        for(let i=0; i<words.length; i++){
            dfs(str+words[i])
        }
    }

    dfs("");
    return answer
}
